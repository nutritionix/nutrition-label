	'showMonoFat' : false,

	'valueCalories' : 450,
	'valueFatCalories' : 430,
	'valueTotalFat' : 48,
	'valueSatFat' : 6,
	'valueTransFat' : 0,
	'valueCholesterol' : 30,
	'valueSodium' : 780,
	'valueTotalCarb' : 3,
	'valueFibers' : 0,

	'valueProteins' : 3,
	'valueVitaminA' : 0,
